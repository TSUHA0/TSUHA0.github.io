#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 10;
int q[N];

void quick_sort(int q[], int l, int r) {
    if (l >= r) return;                                 // 边界条件，注意！！
    int x = q[l + r >> 1], i = l - 1, j = r + 1;
    while (i < j) {
        while(q[++i] < x);
        while(q[--j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> q[i];
    quick_sort(q, 0, n - 1);
    for (int i = 0; i < n; ++i) cout << q[i] << " ";
    return 0;
}

class Solution {
public:
    vector<int> tmp;

    void merge_sort(vector<int> &nums, int l, int r) {
        if (l >= r) return;
        int mid = l + r >> 1;
        merge_sort(nums, l, mid), merge_sort(nums, mid + 1, r);
        int i = l, j = mid + 1, k = 0;                               // i的边界赋值需要注意！！不是0，是l
        while (i <= mid && j <= r) {
            if (nums[i] <= nums[j]) tmp[k++] = nums[i++];
            else tmp[k++] = nums[j++];
        }
        while(i <= mid) tmp[k++] = nums[i++];
        while(j <= r) tmp[k++] = nums[j++];
        for (int i = l, j = 0; i <= r; ++i, ++j) nums[i] = tmp[j];
    }

    vector<int> sortArray(vector<int>& nums) {
        tmp.resize(nums.size());
        merge_sort(nums, 0, nums.size() - 1);
        return nums;
    }
};

int q[N], cnt[N];

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> q[i];
    for (int i = 0; i < m; i++) {
        int l, r, c;
        cin >> l >> r >> c;
        cnt[l] += c;							// 注意是 += 赋值
        cnt[r + 1] -= c;
    }
    int curr = 0;
    for (int i = 1; i <= n; i ++) {
        curr += cnt[i];
        q[i] += curr;
    }
    for (int i = 1; i <= n; i ++) cout << q[i] << " ";
    return 0;
}

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int l = 0, r = -1, ret = 0;
        map<char, bool> dic;
        while (++ r < s.size()) {
            if(dic[s[r]]) {
                while (++l <= r) {
                    dic[s[l - 1]] = false;
                    if (s[l - 1] == s[r]) break;
                } 
            }
            dic[s[r]] = true;
            // printf("r:%d, l:%d \n", r, l);
            ret = max(ret, r - l + 1);
        }
        return ret;
    }
};

#include <bits/stdc++.h>
using namespace std;

const int M = 1010;
int A[M];

int main() {
    int n;
    cin >> n;
    while (n-- > 0) {
        int num, m;
        cin >> num >> m;
        for (int i = 0; i < m; i ++) cin >> A[i];
        
        // 假设n个人，选A[K] 去求最后剩下的坐标为 f(n, k)
        // 则假设第 A[k] 个人的坐标重置为0
        // 现在轮到 n - 1 个人， 选 A[k + 1] 去求最后剩下的坐标为 f(n - 1, k + 1)
        // 这个坐标和实际的坐标相差 A[k]， 也就是 f(n - 1, k + 1) 和 f(n, k) 相差 A[k]
        // -> f(n, k) = （f(n - 1, k + 1) + A[k]）% n
        
        int x = 0;
        // 假设就剩下1个人，他的下标是0，然后开始计算剩下2个人的情况
        for (int i = 2; i <= num; i ++) {
            x = ( x + A[(num - i) % m] ) % i;
        }
        cout << x << endl;
    }
    return 0;
}
